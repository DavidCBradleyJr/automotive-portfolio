```javascript
async function triggerRebuild() {
  try {
    const token = await getToken();
    await fetch('/.netlify/functions/trigger-rebuild', {
      method: 'POST',
      headers: { Authorization: `Bearer ${token}` },
    });
  } catch (err) {
    console.error('Rebuild trigger failed:', err);
  }
}
```
